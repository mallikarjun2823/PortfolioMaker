portfolio.serializers

DRF serializers for the portfolio app.

This module provides Django REST Framework serializers used by the portfolio API: authentication helpers and model serializers for Portfolio, Project, Skill, Experience, Section, Block and Element. All public serializers use straightforward field declarations and the write serializers include validation helpers used by the service layer.

347class _BaseSectionWriteSerializer(serializers.ModelSerializer):
348    """Base write serializer for Section with shared validators."""
349
350    def validate_name(self, value: str) -> str:
351        """Validate and normalize the section name.
352
353        Args:
354            value: Raw name value.
355
356        Returns:
357            Normalized name string.
358
359        Raises:
360            serializers.ValidationError: If the name is empty.
361        """
362        name = "" if value is None else str(value)
363        name = name.strip()
364        if not name:
365            raise serializers.ValidationError("Name cannot be empty.")
366        return name
367
368    def validate_order(self, value: int) -> int:
369        """Ensure the order value is a positive integer or None for PATCH.
370
371        Args:
372            value: Order value to validate.
373
374        Returns:
375            The validated integer or None.
376        """
377        if value is None:
378            return value
379
380        try:
381            order = int(value)
382        except (TypeError, ValueError):
383            raise serializers.ValidationError("Order must be an integer.")
384
385        # Sections are typically 1-based in the seeded dataset and UI ordering.
386        if order < 1:
387            raise serializers.ValidationError("Order must be >= 1.")
388        return order
389
390    def validate_config(self, value: Any) -> Dict[str, Any]:
391        """Validate the section config is a mapping (dict).
392
393        Args:
394            value: Config value.
395
396        Returns:
397            The validated config dict.
398
399        Raises:
400            serializers.ValidationError: If config is missing or not a dict.
401        """
402        if value is None:
403            raise serializers.ValidationError("Config must be an object.")
404        if not isinstance(value, dict):
405            raise serializers.ValidationError("Config must be an object.")
406        return value

532class _BaseElementWriteSerializer(serializers.ModelSerializer):
533    """Base serializer for creating/updating Element mappings with validators."""
534
535    def validate_label(self, value: str) -> str:
536        """Ensure the element label is present and non-empty."""
537        label = "" if value is None else str(value)
538        label = label.strip()
539        if not label:
540            raise serializers.ValidationError("Label cannot be empty.")
541        return label
542
543    def validate_data_source(self, value: str) -> str:
544        """Normalize and validate the data source token."""
545        ds = "" if value is None else str(value)
546        ds = ds.strip().upper()
547        if not ds:
548            raise serializers.ValidationError("Data source cannot be empty.")
549
550        allowed = {c for c, _ in Element.DataSource.choices}
551        if ds not in allowed:
552            raise serializers.ValidationError("Invalid data source.")
553        return ds
554
555    def validate_field(self, value: str) -> str:
556        """Validate the target field token exists for supported data fields."""
557        f = "" if value is None else str(value)
558        f = f.strip()
559        if not f:
560            raise serializers.ValidationError("Field cannot be empty.")
561
562        allowed = {c for c, _ in Element.DataField.choices}
563        if f not in allowed:
564            raise serializers.ValidationError("Invalid field.")
565        return f
566
567    def validate_order(self, value: int) -> int:
568        """Validate element ordering value (positive integer or None)."""
569        if value is None:
570            return value
571
572        try:
573            order = int(value)
574        except (TypeError, ValueError):
575            raise serializers.ValidationError("Order must be an integer.")
576
577        if order < 1:
578            raise serializers.ValidationError("Order must be >= 1.")
579        return order
580
581    def validate_config(self, value: Any) -> Dict[str, Any]:
582        """Validate that element config is a mapping when provided."""
583        if value is None:
584            raise serializers.ValidationError("Config must be an object.")
585        if not isinstance(value, dict):
586            raise serializers.ValidationError("Config must be an object.")
587        return value
588
589    def validate(self, attrs: Dict[str, Any]) -> Dict[str, Any]:
590        """Cross-field validation to ensure the chosen field matches the data source.
591
592        The method supports PATCH semantics by falling back to instance values
593        when partial attributes are provided.
594        """
595        data = dict(attrs)
596
597        ds = data.get("data_source")
598        f = data.get("field")
599
600        # Support PATCH where only one of the fields is present.
601        if ds is None and getattr(self.instance, "data_source", None) is not None:
602            ds = self.instance.data_source
603        if f is None and getattr(self.instance, "field", None) is not None:
604            f = self.instance.field
605
606        if ds and f:
607            allowed = _ALLOWED_ELEMENT_FIELDS_BY_SOURCE.get(str(ds).upper())
608            if allowed is not None and str(f) not in allowed:
609                raise serializers.ValidationError({"field": "Invalid field for the selected data source."})
610
611        return attrs
