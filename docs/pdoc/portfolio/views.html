 65class PortfolioRenderAPIView(APIView):
 66    """Render an authenticated user's portfolio payload.
 67
 68    Returns a fully-serialized representation including theme, sections and
 69    block items. Requires authentication and only renders portfolios owned
 70    by the requesting user.
 71    """
 72
 73    permission_classes = [IsAuthenticated]
 74    service = PortfolioRenderService()
 75
 76    def get(self, request, portfolio_id: int):
 77        """Return the rendered portfolio payload for the authenticated user.
 78
 79        Args:
 80            request: DRF request with authenticated user.
 81            portfolio_id: ID of the portfolio to render.
 82
 83        Returns:
 84            HTTP 200 with serialized portfolio payload, or 404 if not found.
 85        """
 86        payload = self.service.render_portfolio(
 87            portfolio_id=portfolio_id, user_id=request.user.id, include_unpublished=True
 88        )
 89        return Response(payload, status=status.HTTP_200_OK)

119    authentication_classes = []
120    permission_classes = [AllowAny]
121
122    def post(self, request):
123        """Register a new user.
124
125        Validates input via :class:`RegisterSerializer` then delegates to
126        :class:`AuthService` to create the user and issue JWT tokens.
127        """
128        serializer = RegisterSerializer(data=request.data)
129        serializer.is_valid(raise_exception=True)
130        tokens = AuthService().register(**serializer.validated_data)
131        response = AuthResponseSerializer(tokens)
132        return Response(response.data, status=status.HTTP_201_CREATED)
